#Resource exhaustion in Zimbabwe
country_event = {
	id = flavor_zim.1
	title = flavor_zim.1.t
	desc = flavor_zim.1.d
	picture = AFRICAN_CARAVAN_eventPicture

	fire_only_once = yes

	major = yes

	trigger = {
		tag = GZI
		owns_core_province = 1184
	}

	mean_time_to_happen = {
		days = 30
	}

	option = {
		name = flavor_zim.1.a
		add_stability = -1
		1184 = {
			add_permanent_province_modifier = {
				name = zim_declining_city
				duration = -1
			}
		}
		while = {
			limit = {
				1184 = {
					base_tax = 2
				}
			}
			1184 = { add_base_tax = -1 }
		}
		while = {
			limit = {
				1184 = {
					base_production = 2
				}
			}
			1184 = { add_base_production = -1 }
		}
		while = {
			limit = {
				1184 = {
					base_manpower = 2
				}
			}
			1184 = { add_base_manpower = -1 }
		}
		tooltip = {
			every_owned_province = {
				limit = {
					is_core = ROOT
				}
				add_core = ZIM
			}
		}
		custom_tooltip = flavor_zim.1.a.tt

		hidden_effect = {
			set_global_flag = zim_zimbabwe_mass_movement
			ZIM = {
				country_event = {
					id = ipeZIM.1
				}
			}
		}
	}
}

#Decide between Muslim and Christian merchants
country_event = {
	id = flavor_zim.5
	title = flavor_zim.5.t
	desc = flavor_zim.5.desc
	picture = AFRICAN_EUROPEAN_DEBATE_eventPicture

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			random_known_country = {
				limit = {
					capital_scope = {
						continent = europe
					}
					OR = {
						alliance_with = ROOT
						has_opinion = {
							who = ROOT
							value = 75
						}
						has_guaranteed = ROOT
						overlord_of = ROOT
						is_subject_of = ROOT
					}
					religion_group = christian
				}
				save_event_target_as = zim_european_country
			}
			random_known_country = {
				limit = {
					total_development = 50
					capital_scope = {
						continent = africa
					}
					knows_country = ROOT
					religion_group = muslim
				}
				save_event_target_as = zim_muslim_country
			}
		}
	}

	option = {	#Christian traders
		name = flavor_zim.5.a
		custom_tooltip = flavor_zim.5.a.tt
		hidden_effect = {
			export_to_variable = {
				which = zim_new_ruler_religion
				value = religion
				who = event_target:zim_european_country
			}
			set_country_flag = zim_sided_with_christian_traders
		}
		reverse_add_opinion = {
			who = event_target:zim_european_country
			modifier = zim_preferred_by_mutapa
		}
		reverse_add_opinion = {
			who = event_target:zim_muslim_country
			modifier = zim_refused_by_mutapa
		}

		ai_chance = {
			factor = 1
		}
	}

	option = {	#Muslim traders
		name = flavor_zim.5.b
		custom_tooltip = flavor_zim.5.b.tt
		hidden_effect = {
			export_to_variable = {
				which = zim_new_ruler_religion
				value = religion
				who = event_target:zim_muslim_country
			}
			set_country_flag = zim_sided_with_muslim_traders
		}
		reverse_add_opinion = {
			who = event_target:zim_muslim_country
			modifier = zim_preferred_by_mutapa
		}
		reverse_add_opinion = {
			who = event_target:zim_european_country
			modifier = zim_refused_by_mutapa
		}

		ai_chance = {
			factor = 1
		}
	}

	option = {	#Balanced
		name = flavor_zim.5.c
		add_dip_power = 50
		reverse_add_opinion = {
			who = event_target:zim_muslim_country
			modifier = zim_exploited_merchants
		}
		reverse_add_opinion = {
			who = event_target:zim_european_country
			modifier = zim_exploited_merchants
		}

		ai_chance = {
			factor = 1
		}
	}

	option = {	#Do I look like I care?
		name = flavor_zim.5.d
		add_mercantilism = 5
		add_stability = -1

		ai_chance = {
			factor = 1
		}
	}
}

#Accidental crusade / jihad
country_event = {
	id = flavor_zim.6
	title = flavor_zim.6.t
	desc = {
		desc = flavor_zim.6.d.a
		trigger = {
			has_country_flag = zim_sided_with_christian_traders
		}
	}
	desc = {
		desc = flavor_zim.6.d.b
		trigger = {
			has_country_flag = zim_sided_with_muslim_traders
		}
	}
	picture = AFRICAN_CHIEFTAIN_eventPicture
	fire_only_once = yes
	
	trigger = {
		OR = {
			has_country_flag = zim_sided_with_christian_traders
			has_country_flag = zim_sided_with_muslim_traders
		}
	}

	mean_time_to_happen = {
		months = 60
	}

	immediate = {
		hidden_effect = {
			set_ruler_religion = variable:zim_new_ruler_religion
		}
	}

	option = {	#We shall be Christians!
		name = flavor_zim.6.a
		trigger = {
			has_country_flag = zim_sided_with_christian_traders
		}
		add_stability = -3
		change_religion = variable:zim_new_ruler_religion

		ai_chance = {
			factor = 0
		}
	}

	option = {	#We shall be Muslims!
		name = flavor_zim.6.b
		trigger = {
			has_country_flag = zim_sided_with_muslim_traders
		}
		add_stability = -3
		change_religion = variable:zim_new_ruler_religion

		ai_chance = {
			factor = 0
		}
	}

	option = {	#Kill the ruler!
		name = flavor_zim.6.c
		kill_ruler = yes
		add_stability = 3

		ai_chance = {
			factor = 1
		}
	}

	option = {	#Eh whatever
		name = flavor_zim.6.d
		add_legitimacy = -20

		ai_chance = {
			factor = 0
		}
	}

	after = {
		clr_country_flag = zim_sided_with_christian_traders
		clr_country_flag = zim_sided_with_muslim_traders
		capital_scope = {
			change_religion = ROOT
		}
		if = {
			limit = {
				OR = {
					religion = catholic
					religion = reformed
					religion = protestant
					religion = anglican
					religion = hussite
				}
			}
			add_prestige = 100	#You always lose 100 prestige when converting...
		}
	}
}
