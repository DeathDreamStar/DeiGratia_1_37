# The Queen of Cyprus
country_event = {
	id = ipeVEN.1
	title = ipeVEN.1.t
	desc = ipeVEN.1.d
	picture = KING_DOING_GREAT_eventPicture

	fire_only_once = yes

	trigger = {
		tag = CYP
		is_subject = no
		government = monarchy
		is_female = no
		VEN = {
			is_subject = no
			exists = yes
		}
		has_consort = no
		is_year = 1460
	}

	mean_time_to_happen = {
		months = 6
		modifier = {
			factor = 0.5
			alliance_with = VEN
		}
	}

	option = { # Let us take a Venetian bride!
		name = ipeVEN.1.a
		set_country_flag = venetian_inheritance
		define_consort = {
			name = "Catarina"
			dynasty = "Corner"
			country_of_origin = VEN
			religion = VEN
			culture = venetian
			adm = 1
			dip = 1
			mil = 1
		}
		create_marriage = VEN
		VEN = {
			add_opinion = {
				who = CYP
				modifier = accepted_venetian_queen
			}
			hidden_effect = {
				country_event = {
					id = ipeVEN.5
					days = 1
				}
			}
		}
	}

	option = { # Let us marry a local talent instead.
		name = ipeVEN.1.b
		define_consort = {
			country_of_origin = ROOT
			religion = ROOT
			culture = ROOT
		}
	}
}
