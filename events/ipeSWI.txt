# The Seizure of Thurgau
country_event = {
	id = ipeSWI.4
	title = ipeSWI.4.t
	desc = ipeSWI.4.d
	picture = LAND_MILITARY_eventPicture

	is_triggered_only = yes

	option = { # The Confederacy grows stronger.
		name = ipeSWI.1.a
		3744 = {
			owner = {
				add_opinion = {
					who = ROOT
					modifier = seized_thurgau
				}
				hidden_effect = {
					country_event = {
						id = ipeSWI.5
					}
				}
			}
			cede_province = ROOT
			add_core = ROOT
		}
		hidden_effect = {
			if = {
				limit = {
					KNZ = {
						has_reform = free_city
						is_neighbor_of = SWI
					}
				}
				KNZ = {
					country_event = {
						id = ipeSWI.6
						days = 30
					}
				}
			}
		}
	}
}

# Konstanz Petitions to Join the Confederacy

	desc = ipeSWI.7.d

		name = ipeSWI.7.a
		inherit = KNZ
		add_prestige = 5
		add_estate_influence_modifier = {
			estate = estate_burghers
			desc = ADMITTED_KONSTANZ
			influence = 10
			duration = 7300
		}
	}

	option = { # Reject them.
		name = ipeSWI.7.b

		}
		hidden_effect = {
			KNZ = {

					id = ipeSWI.8
				}
			}
		}
	}
}
