country_event = {
	id = ipeARA.3
	title = ipeARA.3.t
	desc = ipeARA.3.d
	picture = ANGRY_MOB_eventPicture

	fire_only_once = yes

	trigger = {
		tag = ARA
		valid_for_personal_unions_trigger = yes
		has_ruler_flag = carlos_de_viana
		owns = 213
		any_owned_province = {
			area = catalonia_area
		}
		ANJ = {
			exists = yes
			valid_for_personal_unions_trigger = yes
		}
	}

	mean_time_to_happen = {
		months = 60
	}

	option = { # Crush them!
		name = ipeARA.3.a
		213 = {
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				leader = "Jean"
				leader_dynasty = "de Valois"
				win = yes
			}
		}
		random_owned_province = {
			limit = {
				area = catalonia_area
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				win = yes
			}
		}
		random_owned_province = {
			limit = {
				area = catalonia_area
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				win = yes
			}
		}
		ANJ = {
			add_casus_belli = {	
				target = ARA
				type = cb_restore_personal_union

			}

		}
	}
}
