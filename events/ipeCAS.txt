	id = ipeCAS.7

	fire_only_once = yes

	trigger = {

		}

	}

	option = { # Let us sign the Treaty of Alcáçovas.

		}

			}
			366 = {
				cede_province = CAS
				remove_core = POR
			}
		}
		if = {
			limit = {

			}

				cede_province = CAS

			}
		}
		add_historical_friend = CAS
		CAS = {
			add_historical_friend = POR

				who = POR

			}
			country_event = {
				id = ipeCAS.8
			}
		}

	}

	option = { # We will never align with Castile.

		}
		tooltip = {

						}
					}

					}
				}
				4565 = {
					add_core = CAS
				}
			}
		}
		clr_country_flag = canaries_rebel
	}
