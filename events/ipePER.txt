# The Rise of Twelverism
country_event = {
	id = ipePER.1
	title = ipePER.1.t
	desc = ipePER.1.d
	picture = RELIGIOUS_WARS_eventPicture

	fire_only_once = yes

	trigger = {
		tag = ARL
	}

	mean_time_to_happen = {
		months = 24
		modifier = {
			factor = 0.1
			is_at_war = yes
		}
	}

	option = { # We will embrace Joneid's teachings.
		name = ipePER.1.a
		ai_chance = {
			factor = 100
		}
		define_heir = {
			name = "Joneid"
			dynasty = "Safavi"
			adm = 3		
			dip = 6
			mil = 4
			age = 30
		}
		kill_ruler = yes
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality = inspiring_leader_personality }
		}
		hidden_effect = {
			change_religion = shiite
			every_owned_province = {
				limit = {
					OR = {
						religion = sunni
						religion = ibadi
					}
				}
				change_religion = shiite
			}
		}
		custom_tooltip = ipePER.1.a.tooltip
		hidden_effect = {
			change_estate_land_share = {
				estate = estate_church
				share = -7.5
			}
			change_estate_land_share = {
				estate = estate_burghers
				share = -7.5
			}
			change_estate_land_share = {
				estate = estate_nobles
				share = -7.5
			}
			change_estate_land_share = {
				estate = estate_dhimmi
				share = -7.5
			}
			set_ai_personality = {
				personality = ai_militarist
				locked = no
			}
		}
	}

	option = { # We reject this heresy!
		name = ipePER.1.b
		ai_chance = {
			factor = 0
		}
		capital_scope = {
			spawn_rebels = {
				type = shiite_rebels
				size = 1
				leader = "Sheikh Joneid"
			}
		}
	}
}

# The Qizilbash Rebellion
country_event = {
	id = ipePER.6
	title = ipePER.6.t
	desc = ipePER.6.d
	picture = RELIGIOUS_WARS_eventPicture

	is_triggered_only = yes

	immediate = {
		random_owned_province = {
			limit = {
				is_capital = no
				OR = {
					religion = sunni
					religion = shiite
					religion = ibadi
				}
			}
			set_province_flag = qizilbash_rebels_1
		}
		random_owned_province = {
			limit = {
				is_capital = no
				OR = {
					religion = sunni
					religion = shiite
					religion = ibadi
				}
				NOT = {
					has_province_flag = qizilbash_rebels_1
				}
			}
			set_province_flag = qizilbash_rebels_2
		}
		random_owned_province = {
			limit = {
				is_capital = no
				OR = {
					religion = sunni
					religion = shiite
					religion = ibadi
				}
				NOT = {
					has_province_flag = qizilbash_rebels_1
					has_province_flag = qizilbash_rebels_2
				}
			}
			set_province_flag = qizilbash_rebels_3
		}
	}

	option = { # We will crush them!
		name = ipePER.6.a
		if = {
			limit = {
				NOT = {
					tag = TUR
				}
			}
			capital_scope = {
				spawn_rebels = {
					type = shiite_rebels
					size = 1
				}
			}
		}
		random_owned_province = {
			limit = {
				has_province_flag = qizilbash_rebels_1
			}
			spawn_rebels = {
				type = shiite_rebels
				size = 1
			}
			clr_province_flag = qizilbash_rebels_1
		}
		random_owned_province = {
			limit = {
				has_province_flag = qizilbash_rebels_2
			}
			spawn_rebels = {
				type = shiite_rebels
				size = 1
			}
			clr_province_flag = qizilbash_rebels_2
		}
		random_owned_province = {
			limit = {
				has_province_flag = qizilbash_rebels_3
			}
			spawn_rebels = {
				type = shiite_rebels
				size = 1
			}
			clr_province_flag = qizilbash_rebels_3
		}
		if = {
			limit = {
				NOT = {
					tag = TUR
				}
			}
			add_country_modifier = {
				name = disloyal_qizilbash
				duration = 3650
			}
		}
	}
}
