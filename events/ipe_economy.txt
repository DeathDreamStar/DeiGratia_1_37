# Commercial Ventures
country_event = {
	id = ipe_economy.9
	title = ipe_economy.9.t
	desc = ipe_economy.9.d
	picture = MERCHANTS_TALKING_eventPicture

	trigger = {
		OR = {
			AND = {
				NOT = {
					total_development = 100
				}
				OR = {
					has_country_modifier = growing_commercial_ventures_100
					has_country_modifier = growing_commercial_ventures_300
					has_country_modifier = growing_commercial_ventures_1000
				}
			}
			AND = {
				total_development = 100
				NOT = {
					total_development = 300
				}
				NOT = {
					has_country_modifier = growing_commercial_ventures_100
				}
			}
			AND = {
				total_development = 300
				NOT = {
					total_development = 1000
				}
				NOT = {
					has_country_modifier = growing_commercial_ventures_300
				}
			}
			AND = {
				total_development = 1000
				NOT = {
					has_country_modifier = growing_commercial_ventures_1000
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 6
	}

	option = { # We will soon see how this affects us.
		name = ipe_economy.9.a
		if = {
			limit = {
				NOT = {
					total_development = 100
				}
				OR = {
					has_country_modifier = growing_commercial_ventures_100
					has_country_modifier = growing_commercial_ventures_300
					has_country_modifier = growing_commercial_ventures_1000
				}
			}
			remove_country_modifier = growing_commercial_ventures_100
			remove_country_modifier = growing_commercial_ventures_300
			remove_country_modifier = growing_commercial_ventures_1000

				}

					has_country_modifier = growing_commercial_ventures_300
