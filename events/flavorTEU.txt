# Prussian Confederation rises up
country_event = {
	id = flavor_teu.1
	title = "flavor_teu.EVTNAME1"
	desc = "flavor_teu.EVTDESC1"
	picture = BATTLE_VISTULA_LAGOON_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		NOT = { is_year = 1500 }
		tag = TEU
		NOT = { has_country_flag = pru_confederation_curtailed }
		POL = {
			NOT = {
				truce_with = TEU
			}
		}
		has_country_flag = pru_confederation_flag
		is_subject = no
		exists = POL
		is_neighbor_of = POL
		NOT = { alliance_with = POL }
		POL = { is_subject = no }
		POL = { num_of_cities = 10 }
		num_of_cities = 5
		OR = {
			NOT = { exists = DNZ }
			DNZ = {
				is_subject_of = ROOT
			}
		}
		OR = {
			AND = {
				owns = 43
				controls = 43
			}
			DNZ = { owns = 43 controls = 43 }
		}
	}
	
	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.5
			is_year = 1450
		}
		modifier = {
			factor = 0.5
			is_year = 1460
		}
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = pru_confederation_happened
			1859 = {
				add_core = DNZ
			}
			6413 = {
				add_core = DNZ
			}
			if = {
				limit = {
					NOT = {
						exists = DNZ
					}
				}
				release = DNZ
				DNZ = {
					change_government = republic
					add_government_reform = oligarchy_reform
					adopt_reform_progress = ROOT
				}
				DNZ = {
					capital_scope = {
						build_to_forcelimit = {
							infantry = 0.6
							cavalry = 0.4
						}
					}
					random_owned_province = {
						limit = {
							has_port = yes
						}
						build_to_forcelimit = {
							galley = 0.4
							light_ship = 0.4
							transport = 0.2
						}
					}
				}
			}
			42 = {
				add_core = DNZ
			}
			43 = {
				add_core = DNZ
			}
			1841 = {
				add_core = DNZ
			}
			2997 = {
				add_core = DNZ
			}
			3923 = {
				add_core = DNZ
			}
			3924 = {
				add_core = DNZ
			}
			6253 = {
				add_core = DNZ
			}
			POL = {
				vassalize = DNZ
			}
		}
	}
	
	option = { # Go to war
		name = "flavor_teu.1.a"
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				OR = {
					is_bankrupt = yes
					war_exhaustion = 10
				}
			}
		}
		custom_tooltip = flavor_teu.1.tooltip
		hidden_effect = {
			POL = {
				country_event = { id = flavor_teu.2 }
			}
			remove_estate_privilege = estate_burghers_prussian_confederation
		}
	}
	option = { # Let them go
		name = "flavor_teu.1.b"
		ai_chance = { 
			factor = 100 
			modifier = {
				factor = 0
				OR = {
					is_bankrupt = no
					NOT = { war_exhaustion = 10 }
				}
			}
		}
		add_stability = -1
		add_prestige = -20
		hidden_effect = {
			remove_estate_privilege = estate_burghers_prussian_confederation
		}
	}
}

		}
	}

	}
	
	option = {
		name = flavor_teu.5.a
