# The Reunification of Morocco
country_event = {
	id = ipeMOR.2
	title = ipeMOR.2.t
	desc = ipeMOR.2.d
	picture = CONQUEST_eventPicture

	is_triggered_only = yes

	option = { # Morocco shall be one!
		name = ipeMOR.2.a
		if = {
			limit = {
				MRK = {
					is_subject_of = ROOT
					NOT = {
						num_of_cities = 10
					}
				}
			}
			inherit = MRK
		}
		if = {
			limit = {
				TFL = {
					is_subject_of = ROOT
					NOT = {
						num_of_cities = 10
					}
				}
			}
			inherit = TFL
		}
		if = {
			limit = {
				SOS = {
					is_subject_of = ROOT
					NOT = {
						num_of_cities = 10
					}
				}
			}
			inherit = SOS
		}
		if = {
			limit = {
				DBD = {
					is_subject_of = ROOT
					NOT = {
						num_of_cities = 10
					}
				}
			}
			inherit = DBD
		}
		add_prestige = 20
		add_piety = -0.1
		set_global_flag = reunited_morocco
	}
}
