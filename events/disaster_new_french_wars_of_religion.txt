# The Surprise of Meaux
country_event = {
	id = new_french_wars_of_religion.1
	title = new_french_wars_of_religion.1.t
	desc = {
		trigger = {
			religion = catholic
		}
		desc = new_french_wars_of_religion.1.d.catholic
	}
	desc = {
		trigger = {
			NOT = { religion = catholic }
		}
		desc = new_french_wars_of_religion.1.d.protestant
	}
	picture = BATTLE_eventPicture

	fire_only_once = yes

	is_triggered_only = yes

	trigger = {
		has_disaster = french_wars_of_religion
		OR = {
			AND = {
				religion = catholic
				any_owned_province = {
					OR = {
						religion = protestant
						religion = reformed
					}
				}
			}
			AND = {
				NOT = {
					religion = catholic
				}
				any_owned_province = {
					religion = catholic
				}
			}
		}
	}

	option = { # Their sedition will be punished...
		name = new_french_wars_of_religion.1.a
		set_country_flag = bartholomew_massacre_prelude
		add_stability = -1
		if = {
			limit = {
				religion = catholic
				any_owned_province = {
					religion = protestant
				}
			}
			random_owned_province = {
				limit = {
					religion = protestant
					is_capital = no
				}
				spawn_rebels = {
					type = protestant_rebels
					size = 2
				}
			}
		}
		if = {
			limit = {
				religion = catholic
				any_owned_province = {
					religion = reformed
				}
			}
			random_owned_province = {
				limit = {
					religion = reformed
					is_capital = no
				}
				spawn_rebels = {
					type = reformed_rebels
					size = 2
				}
			}
		}
		if = {
			limit = {
				NOT = {
					religion = catholic
				}
			}
			random_owned_province = {
				limit = {
					religion = catholic
					is_capital = no
				}
				spawn_rebels = {
					type = catholic_rebels
					size = 2
				}
				set_province_flag = catholic_rebel_spawn
			}
			random_owned_province = {
				limit = {
					religion = catholic
					is_capital = no
					NOT = {
						has_province_flag = catholic_rebel_spawn
					}
				}
				spawn_rebels = {
					type = catholic_rebels
					size = 2
				}
			}
		}
	}

	after = {
		every_owned_province = {
			limit = {
				has_province_flag = catholic_rebel_spawn
			}
			clr_province_flag = catholic_rebel_spawn
		}
	}
}

	desc = {
		trigger = {
			has_heir = yes
		}
		desc = new_french_wars_of_religion.2.d.has_heir
	}
	desc = {
		trigger = {
			has_heir = no
		}
		desc = new_french_wars_of_religion.2.d.no_heir
	}
	picture = DEATH_OF_HEIR_eventPicture

	fire_only_once = yes

	is_triggered_only = yes

	trigger = {
		has_disaster = french_wars_of_religion
		has_country_flag = henry_of_navarre
		OR = {
			religion = catholic
			religion = protestant
			religion = reformed
		}
	}

	immediate = {
		clr_country_flag = henry_of_navarre
	}

	option = { # Our faith is in France. The monarchy rules from blood.
		name = new_french_wars_of_religion.2.a
		trigger = {
			religion = catholic
		}
		custom_tooltip = new_french_wars_of_religion.2.a.catholic.tooltip
		if = {
			limit = {
				has_heir = yes
			}
			kill_heir = { allow_new_heir = no }
		}
		set_country_flag = catholic_league_rising
		add_country_modifier = {
			name = league_rising
			duration = -1
			hidden = yes
		}
	}

	option = { # Our faith is in France. The monarchy rules from blood.
		name = new_french_wars_of_religion.2.a
		trigger = {
			OR = {
				religion = protestant
				religion = reformed
			}
		}
		custom_tooltip = new_french_wars_of_religion.2.a.protestant.tooltip
		if = {
			limit = {
				has_heir = yes
			}
			kill_heir = { allow_new_heir = no }
		}
		set_country_flag = protestant_league_rising
		add_country_modifier = {
			name = league_rising
			duration = -1
			hidden = yes
		}
	}

	after = {
		if = {
			limit = {
				has_heir = yes
			}
			remove_heir = {}
		}
	}
}

# The Catholic League Declares War!
country_event = {
	id = new_french_wars_of_religion.8
	title = new_french_wars_of_religion.8.t
	desc = new_french_wars_of_religion.8.d
	picture = RELIGIOUS_WARS_eventPicture

	fire_only_once = yes

	major = yes

	trigger = {
		has_country_flag = war_with_catholic_league
	}

	mean_time_to_happen = {
		days = 30
	}

	immediate = {
		clr_country_flag = war_with_catholic_league
		add_stability = -2
		hidden_effect = {
			every_owned_province = {
				limit = {
					religion = catholic
					continent = europe
					OR = {
						culture = cosmopolitan_french
						culture = champenois
						culture = picard
						culture = normand
						culture = burgundian
						culture = occitain
						culture = wallonian
						culture = lorrainer
						culture = arpitan
					}
				}
				add_core = CTL
			}
			release = CTL
			if = {
				limit = {
					183 = {
						religion = catholic
					}
				}
				183 = {
					cede_province = CTL
				}
				CTL = {
					set_capital = 183
				}
			}
			CTL = {
				if = {
					limit = {
						FRA = {
							government_rank = 3
						}
					}
					set_government_rank = 3
				}
				if = {
					limit = {
						FRA = {
							NOT = {
								government_rank = 3
							}
							government_rank = 2
						}
					}
					set_government_rank = 2
				}
				if = {
					limit = {
						FRA = {
							NOT = {
								government_rank = 2
							}
						}
					}
					set_government_rank = 1
				}
				change_religion = catholic
				change_government = monarchy
				add_government_reform = feudalism_reform
				adopt_reform_progress = ROOT
				define_ruler = {
					name = "Charles"
					dynasty = "de Bourbon"
					adm = 1
					dip = 3
					mil = 1
				}
				if = {
					limit = {
						has_dlc = "Rights of Man"
					}
					hidden_effect = { add_ruler_personality = zealot_personality }
				}
				capital_scope = {
					build_to_forcelimit = {
						infantry = 1.6
						cavalry = 0.4
						artillery = 0.4
					}
				}
				random_owned_province = {
					limit = {
						has_port = yes
					}
					build_to_forcelimit = {
						heavy_ship = 0.5
						light_ship = 0.5
						transport = 0.5
					}
				}
				add_manpower = 10
				add_treasury = 250
			}
			every_owned_province = {
				add_core = CTL
			}
			every_subject_country = {
				limit = {
					capital_scope = {
						region = france_region
					}
					religion = catholic
				}
				random = {
					chance = 50
					CTL = {
						vassalize = PREV
					}
				}
			}
			declare_war_with_cb = {
				who = CTL
				casus_belli = cb_civil_war
			}
		}
	}

	option = { # Grande Dieu Sauve le Roi!
		name = new_french_wars_of_religion.8.a
		tooltip = {
			declare_war_with_cb = {
				who = CTL
				casus_belli = cb_civil_war
			}
		}
		add_yearly_manpower = -3
		remove_country_modifier = edit_dunion
		hidden_effect = {
			remove_country_modifier = league_rising
		}
		remove_country_modifier = french_moderates
		remove_country_modifier = french_fanaticism
		if = {
			limit = {
				any_owned_province = {
					religion = reformed
				}
			}
			define_heir = {
				dynasty = "de Bourbon"
				adm = 3
				dip = 3
				mil = 3
				religion = reformed
				block_disinherit = yes
				claim = 50
				age = 40
			}
			set_heir_flag = worth_a_mass
		}
		if = {
			limit = {
				NOT = {
					any_owned_province = {
						religion = reformed
					}
				}
			}
			define_heir = {
				dynasty = "de Bourbon"
				adm = 3
				dip = 3
				mil = 3
				religion = protestant
				block_disinherit = yes
				claim = 50

			}
			set_heir_flag = worth_a_mass
		}
	}

	option = { # Play as the Catholic League.
		name = new_french_wars_of_religion.8.b
		hidden_effect = { 
			add_yearly_manpower = -3
			remove_country_modifier = edit_dunion
			hidden_effect = {
				remove_country_modifier = league_rising
			}
			remove_country_modifier = french_moderates
			remove_country_modifier = french_fanaticism
			if = {
				limit = {
					any_owned_province = {
						religion = reformed
					}
				}
				define_heir = {
					dynasty = "de Bourbon"
					adm = 3

					claim = 50

				set_heir_flag = worth_a_mass
			}

# The Governor of Brittany
country_event = {
	id = new_french_wars_of_religion.16
	title = new_french_wars_of_religion.16.t
	desc = new_french_wars_of_religion.16.d
	picture = SIEGE_eventPicture

	fire_only_once = yes

	trigger = {
		tag = FRA
		OR = {
			AND = {
				war_with = CTL
				172 = {
					religion = catholic
				}
			}
			AND = {
				war_with = HUG
				172 = {
					OR = {
						religion = protestant
						religion = reformed
					}
				}
			}
		}
		NOT = {
			exists = BRI
		}
		brittany_area = {
			type = all
			is_core = BRI
			owned_by = ROOT
		}
		western_brittany_area = {
			type = all
			is_core = BRI
			owned_by = ROOT
		}
		vannetais_area = {
			type = all
			is_core = BRI
			owned_by = ROOT
		}
		rennais_area = {
			type = all
			is_core = BRI
			owned_by = ROOT
		}
	}

	mean_time_to_happen = {
		days = 30
	}

	immediate = {
		hidden_effect = {
			release = BRI
			BRI = {
				change_government = monarchy
				add_government_reform = feudalism_reform
				adopt_reform_progress = ROOT
				define_ruler = {
					name = "Philippe Emmanuel"
					dynasty = "de Lorraine"
					adm = 4
					dip = 4
					mil = 4
				}
				if = {
					limit = {
						has_dlc = "Rights of Man"
					}
					hidden_effect = { add_ruler_personality = zealot_personality }
				}
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.8
						cavalry = 0.2
						artillery = 0.2
					}
				}
				random_owned_province = {
					limit = {
						has_port = yes
					}
					build_to_forcelimit = {
						heavy_ship = 0.3
						light_ship = 0.3
						transport = 0.3
					}
				}
				add_manpower = 5
				add_treasury = 100
			}
			declare_war_with_cb = {
				who = BRI
				casus_belli = cb_civil_war
			}
		}
	}

	option = { # Brittany will return to the fold!
		name = new_french_wars_of_religion.16.a
		tooltip = {
			declare_war_with_cb = {
				who = BRI
				casus_belli = cb_civil_war
			}
		}
	}

	option = { # Play as Brittany.
		name = new_french_wars_of_religion.16.b
		switch_tag = BRI
	}
}
