# Imperial Mediatization
country_event = {
	id = ipeHRE.1
	title = ipeHRE.1.t
	desc = ipeHRE.1.d
	picture = HRE_eventPicture

	fire_only_once = yes

	trigger = {
		government = monarchy
		is_part_of_hre = yes
		any_neighbor_country = {
			is_part_of_hre = yes
			OR = {
				has_reform = free_city
				government = theocracy
			}
			NOT = {
				total_development = 50
			}
			is_elector = no
			ai = yes
			NOT = { # escaped mediatization
				tag = HAM
				tag = BRE
				tag = HSA
				tag = FRN
			}
		}
		any_known_country = {
			is_revolution_target = yes
			capital_scope = {
				continent = europe
			}
		}
		NOT = { active_imperial_incident = any }
		has_dlc = "Emperor"
	}

	mean_time_to_happen = {
		months = 120
	}

	option = { # It will be a time of great change in the Empire.
		name = ipeHRE.1.a
		set_imperial_incident = incident_mediatization
	}
}

# The Augsburg Confession (Catholic Free Cities)
country_event = {
	id = ipeHRE.8
	title = ipeHRE.5.t
	desc = ipeHRE.8.d
	picture = RELIGION_eventPicture

	trigger = {
		any_known_country = {
			has_country_flag = schmalkaldic_league_leader
			exists = yes
		}
		is_part_of_hre = yes
		religion = catholic
		has_reform = free_city
		NOT = {
			has_country_flag = subscribed_to_augsburg_confession
			has_country_flag = rejected_augsburg_confession
		}
		is_subject = no
	}

	mean_time_to_happen = {
		months = 240
	}

	option = { # Subscribe to the Augsburg Confession.
		name = ipeHRE.6.a
		set_country_flag = subscribed_to_augsburg_confession
		if = {
			limit = {
				capital_scope = {
					NOT = {
						religion = protestant
					}
				}
			}
			capital_scope = {
				change_religion = protestant
			}
		}
		change_religion = protestant
		random_known_country = {
			limit = {
				has_country_flag = schmalkaldic_league_leader
			}
			add_opinion = {
				who = ROOT
				modifier = opinion_grateful
			}
			create_subject = {
				who = ROOT
				subject_type = league_associate
			}
			country_event = {
				id = ipeHRE.7
			}
		}
		emperor = {
			add_opinion = {
				who = ROOT
				modifier = opinion_irritated
			}
		}
		set_country_flag = schmalkaldic_league_member
	}

	option = { # Burn these documents and make sure they are never read again.
		name = ipeHRE.8.b
		set_country_flag = rejected_augsburg_confession
		FROM = {
			add_opinion = {
				who = ROOT
				modifier = opinion_irritated
			}
		}
		emperor = {
			add_opinion = {
				who = ROOT
				modifier = opinion_grateful
			}
		}
		add_republican_tradition = -10
	}
}
