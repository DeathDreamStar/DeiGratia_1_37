# The Last Jagiellon
country_event = {
	id = ipePOL.1
	title = ipePOL.1.t
	desc = ipePOL.1.d
	picture = GOOD_WITH_MONARCH_eventPicture

	trigger = {
		OR = {
			tag = POL
			tag = PLC
		}
		NOT = {
			dynasty = "Jagiellon"
		}
		has_regency = no
		has_country_flag = potential_elective_monarchy
		is_subject = no
		NOT = {
			has_reform = states_general_reform
		}
	}
	
	mean_time_to_happen = {
		days = 15
	}

	immediate = {
		random_country = {
			limit = {
				government = monarchy
				NOT = {
					is_rival = ROOT
					war_with = ROOT
				}
				is_lesser_in_union = no
				dynasty = "Jagiellon"
			}
			set_country_flag = jagiellon_inheritor
		}
	}
	
	option = { # Wait for the nobles to convene.
		name = ipePOL.1.a
		if = {
			limit = {
				has_heir = yes
			}
			remove_heir = {}
		}
		hidden_effect = {
			define_ruler = {
				name = "(Interregnum)"
				regency = yes
				max_random_adm = 0
				max_random_dip = 0
				max_random_mil = 0
			}
		}
		custom_tooltip = ipePOL.1.a.tooltip
		clr_country_flag = potential_elective_monarchy
		set_country_flag = likely_elective_monarchy
    }

	option = { # We need a Jagiellon!
		name = ipePOL.1.b
		ai_chance = {
			factor = 100
		}
		trigger = {
			any_known_country = {
				has_country_flag = jagiellon_inheritor
			}
		}
		if = {
			limit = {
				has_heir = yes
			}
			remove_heir = {}
		}
		hidden_effect = {
			define_ruler = {
				name = "(Interregnum)"
				regency = yes
				max_random_adm = 0
				max_random_dip = 0
				max_random_mil = 0
			}
		}
		custom_tooltip = ipePOL.1.b.tooltip
		random_country = {
			limit = {
				has_country_flag = jagiellon_inheritor
			}
			country_event = {
				id = ipePOL.2
				days = 30
			}
		}
	}
}
