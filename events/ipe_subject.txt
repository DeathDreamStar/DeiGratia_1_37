		OR = {
			is_subject_of_type = defensive_vassal

			is_subject_of_type = defensive_puppet_bishopric
			is_subject_of_type = defensive_league_associate
		}
		is_at_war = no
	}

	mean_time_to_happen = {
		days = 30
	}

					subject = ROOT

				}
			}
		}
		if = {
			limit = {
				is_subject_of_type = defensive_march
			}
			overlord = {
				create_subject = {
					subject = ROOT

			limit = {
				is_subject_of_type = defensive_puppet_bishopric

			overlord = {
				create_subject = {
