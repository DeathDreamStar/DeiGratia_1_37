namespace = religious_reforms

# Nahuatl reform
country_event = {
	id = religious_reforms.1
	title = religious_reforms.1.t
	desc = religious_reforms.1.d
	picture = ANGRY_MOB_eventPicture

	is_triggered_only = yes
	
	trigger = {
		has_dlc = "El Dorado"
		religion = nahuatl
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			add_stability = -2
			add_doom = 25 
		}
		if = {
			limit = {
				NOT = {
					has_country_flag = passed_one_reform
					has_country_flag = passed_two_reforms
					has_country_flag = passed_three_reforms
					has_country_flag = passed_four_reforms
				}
			}
			set_country_flag = passed_one_reform
		}
		if = {
			limit = {
				has_country_flag = passed_one_reform
			}
			clr_country_flag = passed_one_reform
			set_country_flag = passed_two_reforms
		}
		if = {
			limit = {
				has_country_flag = passed_two_reforms
			}
			clr_country_flag = passed_two_reforms
			set_country_flag = passed_three_reforms
		}
		if = {
			limit = {
				has_country_flag = passed_three_reforms
			}
			clr_country_flag = passed_three_reforms
			set_country_flag = passed_four_reforms
		}
		if = {
			limit = {
				has_country_flag = passed_four_reforms
			}
			clr_country_flag = passed_four_reforms
		}
	}
	
	option = {
		name = religious_reforms.1.a
		tooltip = {
			add_doom = 25 
			add_stability = -2
		}
	}
}

# Inti reform
country_event = {
	id = religious_reforms.2
	title = religious_reforms.2.t
	desc = religious_reforms.2.d
	picture = CIVIL_WAR_eventPicture

	is_triggered_only = yes
	
	trigger = {
		has_dlc = "El Dorado"
		religion = inti
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {		
			add_authority = -100 
			add_stability = -2
			
			random_owned_province = {
				limit = { 
					is_in_capital_area = yes 
					is_capital = no
				}
				set_province_flag = civil_war_prov_1
			}		
			random_owned_province = {
				limit = { 
					is_in_capital_area = yes
					is_capital = no
					NOT = { has_province_flag = civil_war_prov_1 }
				}
				set_province_flag = civil_war_prov_2
			}
			random_owned_province = {
				limit = { 
					is_in_capital_area = yes 
					is_capital = no
					NOT = { has_province_flag = civil_war_prov_1 }
					NOT = { has_province_flag = civil_war_prov_2 }
				}
				set_province_flag = civil_war_prov_3
			}			
			
			random_owned_province = {
				limit = { has_province_flag = civil_war_prov_1 }
				pretender_rebels = 4
				clr_province_flag = civil_war_prov_1
			}
			random_owned_province = {
				limit = { has_province_flag = civil_war_prov_2 }
				pretender_rebels = 3
				clr_province_flag = civil_war_prov_2
			}
			random_owned_province = {
				limit = { has_province_flag = civil_war_prov_3 }
				pretender_rebels = 3
				clr_province_flag = civil_war_prov_3
			}			
		}
		
	}	
	
	option = {
		name = religious_reforms.2.a
		tooltip = {
			add_stability = -2
			add_authority = -100			
		}		
	}
}
