# Decline in [From.GetName] (Tax)
country_event = {
    id = ipe_development_display.4
    title = ipe_development_display.4.t
    desc = ipe_development_display.4.d
    picture = FAMINE_eventPicture
    goto = from

    is_triggered_only = yes

    option = { # This is unfortunate news.
        name = ipe_development_display.4.a
        tooltip = {
            FROM = {
                add_base_tax = -1
                add_devastation = -5
            }
            if = {
                limit = {
                    FROM = {
                        owner = {
                            any_owned_province = {
                                has_province_flag = tax_refugee_recipient
                            }
                        }
                    }
                }
                FROM = {
                    owner = {
                        random_owned_province = {
                            limit = {
                                has_province_flag = tax_refugee_recipient
                            }
                            add_base_tax = 1
                        }
                    }
                }
            }
            if = {
                limit = {
                    any_neighbor_country = {
                        any_owned_province = {
                            has_province_flag = international_tax_refugee_recipient
                        }
                    }
                }
                random_neighbor_country = {
                    limit = {
                        any_owned_province = {
                            has_province_flag = international_tax_refugee_recipient
                        }
                    }
                    random_owned_province = {
                        limit = {
                            has_province_flag = international_tax_refugee_recipient
                        }
                        add_base_tax = 1
                    }
                }
            }
        }
        FROM = {
            owner = {
                random_owned_province = {
                    limit = {
                        has_province_flag = tax_refugee_recipient
                    }
                    clr_province_flag = tax_refugee_recipient
                }
            }
        }
    }
}

# Old World Diseases
country_event = {
    id = ipe_development_display.7
    title = ipe_development_display.7.t
    desc = ipe_development_display.7.d
    picture = PLAGUE_eventPicture

    is_triggered_only = yes

    option = { # Have mercy upon us all!
        name = ipe_development_display.7.a
        tooltip = {
            if = {
                limit = {
                    FROM = {
                        NOT = {
                            development = 20
                        }
                    }
                }
                FROM = {
                    add_permanent_province_modifier = {
                        name = old_world_diseases_normal_2
                        duration = 1825
                    }
                }
            }
            if = {
                limit = {
                    FROM = {
                        development = 20
                    }
                }
                FROM = {
                    add_permanent_province_modifier = {
                        name = old_world_diseases_bad_2
                        duration = 1825
                    }
                }
            }
        }
    }
}
