        name = ipeALB.2.a

            }
        }
        FROM = {

        }
        add_country_modifier = {
            name = treaty_of_gaeta
            duration = 9125
        }

			}

                country_event = {
                    id = ipeALB.3
                }
            }
        }

    }

    option = { # We do not need the help of foreigners.
        name = ipeALB.2.b
        ai_chance = {
            factor = 20
            modifier = {
                factor = 0
                war_with = TUR
            }
        }
        FROM = {
			add_opinion = {
				who = ROOT
				modifier = refused_treaty_of_gaeta
			}
            hidden_effect = {
                country_event = {
                    id = ipeALB.4
                }	
            }
		}
        add_prestige = 10
    }
}

# The Treaty of Gaeta (Albania)
country_event = {
    id = ipeALB.5
    title = ipeALB.5.t
    desc = ipeALB.5.d
    picture = DIPLOMACY_eventPicture

    fire_only_once = yes

    major = yes

    trigger = {
        has_country_flag = albania_treaty_of_gaeta
        is_subject = no
        is_at_war = no
        AND = {
            northern_albania_area = {
                NOT = {
                    controlled_by = TUR
                    owned_by = TUR
                }
            }
            albania_area = {
                NOT = {
                    controlled_by = TUR
                    owned_by = TUR
                }
            }
        }
        any_known_country = {
            has_country_flag = treaty_of_gaeta_successful
            is_subject = no
            OR = {
                alliance_with = ROOT
                has_guaranteed = ROOT
            }
        }
    }

    mean_time_to_happen = {
		days = 180
	}

    option = { # Let us make good on our promises.
        name = ipeALB.5.a
        ai_chance = {
            factor = 50
        }

        random_known_country = {
            limit = {
                has_country_flag = treaty_of_gaeta_successful
            }
            create_subject = {
                subject_type = vassal
                subject = ROOT
            }
            add_opinion = {
                who = ROOT
                modifier = honored_treaty_of_gaeta
            }
            country_event = {
                id = ipeALB.6
            }
        }
    }

    option = { # Albania shall know no foreign rulers.
        name = ipeALB.5.b
        ai_chance = {
            factor = 50
        }

        remove_country_modifier = treaty_of_gaeta
        add_country_modifier = {
            name = broke_treaty_of_gaeta
            duration = 3650
        }
        random_known_country = {
            limit = {
                has_country_flag = treaty_of_gaeta_successful
            }
            add_opinion = {
                who = ROOT
                modifier = dishonored_treaty_of_gaeta
            }
            remove_opinion = {
                who = ROOT
                modifier = accepted_treaty_of_gaeta
            }
            country_event = {
                id = ipeALB.7
            }
        }
    }
}
