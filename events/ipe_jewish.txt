		}
		modifier = {
			factor = 0.9
			ruler_has_personality_ancestor = { key = kind_hearted }
		}
		modifier = {
			factor = 0.9
			ruler_has_personality_ancestor = { key = righteous }
		}
		modifier = {
			factor = 0.9
			ruler_has_personality_ancestor = { key = tolerant }
		}
		modifier = {
			factor = 0.7
			has_idea_group = humanist_ideas
		}
		modifier = {
			factor = 1.1
			has_idea_group = religious_ideas
		}
		modifier = {
			factor = 0.9
			jewish = 2
		}
		modifier = {
			factor = 1.1
			NOT = {
				jewish = 0
			}
		}
		modifier = {
			factor = 0.5
			has_country_modifier = jews_in_poland
		}
	}

	option = {
		name = ipe_jewish.1.a
		random_owned_province = {
			add_province_modifier = {
				name = encouraged_growth_production
				duration = 18250
			}

		if = {
			limit = {
				religion = confucianism
			}
			add_harmony = 10
		}
		if = {
			limit = {
				religion = shamanism
			}
			unlock_cult = jewish_cult
		}
		add_karma = 25
	}
}
