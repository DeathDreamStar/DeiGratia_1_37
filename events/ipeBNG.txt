	major = yes

	trigger = {
		tag = BNG
		NOT = {
			legitimacy = 50
		}
		government = monarchy
		has_regency = no
	}

	mean_time_to_happen = {
		months = 60
	}

	option = { # Dark times come for Bengal.
		name = ipeBNG.1.a
		kill_ruler = yes

			name = "Shahzada Barbak"
			dynasty = "Habshi"

		}
		hidden_effect = {
			add_ruler_personality = cruel_personality
		}
		set_ruler_flag = barbak_downfall
	}
}
