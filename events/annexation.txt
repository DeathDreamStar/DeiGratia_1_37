# Collapse of the Mamluk Sultanate
country_event = {
	id = annexation.1
	title = annexation.1.t
	desc = annexation.1.d
	picture = BATTLE_eventPicture
	
	fire_only_once = yes

	major = yes

	is_triggered_only = yes

	option = { # We shall rule Egypt directly.
		name = annexation.1.a
		add_adm_power = -100
		add_dip_power = -100
		add_mil_power = -100
		add_prestige = 50
		if = {
			limit = {
				is_in_war = {
					attacker_leader = ROOT
					defender_leader = MAM
				}
			}
			white_peace = MAM
		}
		if = {
			limit = {
				is_in_war = {
					attacker_leader = MAM
					defender_leader = ROOT
				}
			}
			white_peace = MAM
		}
		MAM = {
			remove_historical_rival = ROOT
		}
		remove_historical_rival = MAM
		custom_tooltip = annexation.1.a.tooltip
		hidden_effect = {
			MAM = {
				every_owned_province = {
					limit = {
						is_core = MAM
					}
					cede_province = ROOT
					add_core = ROOT
				}
				every_owned_province = {
					limit = {
						NOT = {
							is_core = MAM
						}
					}
					cede_province = ROOT
				}
				every_core_province = {
					limit = {
						NOT = {
							region = egypt_region
							area = lower_nubia_area
							area = dongola_area
							area = beja_area
							area = upper_nubia_area
							area = sennar_area
							area = shilluk_area
							area = kurdufan_area
							area = darfur_area
						}
					}
					remove_core = MAM
				}
			}
		}
	}

	option = { # Do not infringe on the authority of the local Mamluks.
		name = annexation.1.b
		add_adm_power = -100
		add_dip_power = -100
		add_mil_power = -100
		add_prestige = 50
		if = {
			limit = {
				is_in_war = {
					attacker_leader = ROOT
					defender_leader = MAM
				}
			}
			white_peace = MAM
		}
		if = {
			limit = {
				is_in_war = {
					attacker_leader = MAM
					defender_leader = ROOT
				}
			}
			white_peace = MAM
		}
		MAM = {
			remove_historical_rival = ROOT
		}
		remove_historical_rival = MAM
		custom_tooltip = annexation.1.b.tooltip
		hidden_effect = {
			MAM = {
				every_owned_province = {
					limit = {
						is_core = MAM
						NOT = {
							region = egypt_region
							area = lower_nubia_area
							area = dongola_area
							area = beja_area
							area = upper_nubia_area
							area = sennar_area
						}
					}
					cede_province = ROOT
					add_core = ROOT
				}
				every_owned_province = {
					limit = {
						NOT = {
							is_core = MAM
							region = egypt_region
							area = lower_nubia_area
							area = dongola_area
							area = beja_area
							area = upper_nubia_area
							area = sennar_area
						}
					}
					cede_province = ROOT
				}
				every_subject_country = {
					limit = {
						is_subject_of_type = vassal
						capital_scope = {
							NOT = {
								region = egypt_region
								area = lower_nubia_area
								area = dongola_area
								area = red_sea_coast_area
								area = beja_area
								area = upper_nubia_area
								area = sennar_area
							}
						}
					}
					ROOT = {
						vassalize = PREV
					}
				}
				every_subject_country = {
					limit = {
						is_subject_of_type = march
						capital_scope = {
							NOT = {
								region = egypt_region
								area = lower_nubia_area
								area = dongola_area
								area = red_sea_coast_area
								area = beja_area
								area = upper_nubia_area
								area = sennar_area
							}
						}
					}
					ROOT = {
						create_march = PREV
					}
				}
				every_subject_country = {
					limit = {
						is_subject_of_type = march
						capital_scope = {
							OR = {
								region = egypt_region
								area = lower_nubia_area
								area = dongola_area
								area = red_sea_coast_area
								area = beja_area
								area = upper_nubia_area
								area = sennar_area
								area = shilluk_area
								area = kurdufan_area
								area = darfur_area
							}
						}
					}
					MAM = {
						vassalize = PREV
					}
				}
				every_core_province = {
					limit = {
						NOT = {
							owned_by = MAM
							region = egypt_region
							area = lower_nubia_area
							area = dongola_area
							area = beja_area
							area = upper_nubia_area
							area = sennar_area
							area = shilluk_area
							area = kurdufan_area
							area = darfur_area
						}
					}
					remove_core = MAM
				}
			}
		}
		create_march = MAM
		MAM = {
			add_liberty_desire = -1000
			add_opinion = {
				who = ROOT
				modifier = mamluks_allowed_to_rule
			}
		}
		every_subject_country = {
			limit = {
				NOT = {
					tag = MAM
				}
			}
			add_liberty_desire = -100
		}
	}
}

    id = annexation.3
	title = annexation.3.t
	desc = annexation.3.d

	major = yes

	trigger = {
        tag = AKK
		is_subject = no
		war_with = QAR

			ai = yes

				controlled_by = AKK
			}
		}

			has_global_flag = qara_annexed
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		set_global_flag = qara_annexed
	}

	option = { # The time of the White Sheep has come!
		name = annexation.3.a
		add_adm_power = -100
		add_dip_power = -100
		add_mil_power = -100
		add_prestige = 25
		if = {
			limit = {
				is_in_war = {
					attacker_leader = ROOT
					defender_leader = QAR
				}
			}
			white_peace = QAR
		}
		if = {
			limit = {
				is_in_war = {
					attacker_leader = QAR
					defender_leader = ROOT
				}
			}
			white_peace = QAR
		}
		if = {
			limit = {
				IRQ = {
					is_subject_of = QAR
				}
			}
			QAR = {
				inherit = IRQ
			}
		}
		QAR = {
            every_owned_province = {
				limit = {
					is_core = QAR
				}
				cede_province = ROOT
                add_core = ROOT
			}
            every_owned_province = {
				limit = {
					NOT = {
						is_core = QAR
					}
				}
				cede_province = ROOT
			}
		}
		every_subject_country = {
			add_liberty_desire = -25
		}
		add_yearly_manpower = 5
		persia_region = {
			limit = {
				NOT = {
					is_core = ROOT
				}
				NOT = {
					is_permanent_claim = ROOT
				}
			}
			add_permanent_claim = ROOT
		}
		if = {
			limit = {
				owns = 416
			}
			set_capital = 416
		}
	}
}
