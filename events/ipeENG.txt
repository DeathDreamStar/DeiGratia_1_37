# The Great Fire of London

	id = ipeENG.3
	title = ipeENG.3.t
	desc = ipeENG.3.d
	picture = BURNING_CITY_eventPicture
	goto = 236

	fire_only_once = yes

	major = yes

	trigger = {
		owns = 236
		236 = {
			has_province_flag = great_fire_of_london
			development = 20
			NOT = {
                has_province_modifier = peace_plague
                has_province_modifier = quarantined_plague
			}
		}
	}

	mean_time_to_happen = {
		months = 6
	}

	option = { # London will be reborn more glorious than before...
		name = ipeENG.3.a
		add_stability = -1
		add_years_of_income = -1
		236 = {
			add_base_tax = -2
			add_base_production = -2
			add_base_manpower = -2
			add_province_modifier = {
				name = encouraged_growth_tax
				duration = 18250
			}
			add_province_modifier = {
				name = encouraged_growth_production
				duration = 18250
			}
			add_province_modifier = {
				name = encouraged_growth_manpower
				duration = 18250
			}
			add_permanent_province_modifier = {
				name = the_rebuilding_of_london
				duration = -1
			}
			clr_province_flag = great_fire_of_london
		}
	}
}
