# Western Wrecked Ship (William Adams)
country_event = {
	id = flavor_jap.8
	title = "flavor_jap.EVTNAME8"
	desc = "flavor_jap.EVTDESC8"
	picture = SHIP_SAILING_eventPicture

	fire_only_once = yes
	
	trigger = {
		culture_group = japanese_g
		OR = {
			has_country_modifier = western_trade
			is_incident_active = incident_nanban
			is_incident_happened = incident_nanban
		}
		has_regency = no
		NOT = { has_country_flag = william_adams }
		num_of_ports = 1
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	immediate = {
		set_country_flag = william_adams
	}
	
	option = {
		name = "flavor_jap.EVTOPTA8" # Do nothing
		add_stability_or_adm_power = yes
	}
	option = {
		name = "flavor_jap.EVTOPTB8" # Help the foreign shipwreckers
		add_stability = -1
		add_ruler_modifier = { 
			name = jap_western_wrecked_ship
		}
	}
	option = {
		name = "flavor_jap.EVTOPTC8" # Bring the foreigner to me
		add_prestige = -10
		define_advisor = {
			name = "William Adams"
			culture = english
			religion = protestant
			type = trader
			discount = yes
			skill = 2
		}
	}
}

		name = "flavor_jap.EVTOPTA36"

			limit = { has_country_modifier = ninjas	}
			add_opinion = { who = ROOT modifier = opinion_ninja_attack }
		}
		random_list = {
			50 = { 
				add_prestige = 10
				every_neighbor_country = {
					limit = { has_country_modifier = ninjas	}
					country_event = { id = flavor_jap.37 } 
				}
			}
			50 = { 
				add_prestige = -10
				every_neighbor_country = {
					limit = { has_country_modifier = ninjas	}
					country_event = { id = flavor_jap.38 } 
				} 
			}
		}
		
	}
	option = {
		name = "flavor_jap.EVTOPTB36"

		every_neighbor_country = {
			limit = { has_country_modifier = ninjas	}

		}
	}
}
