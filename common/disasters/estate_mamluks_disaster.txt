estate_mamluks_disaster = {
	potential = {
		has_estate = estate_mamluks
		estate_influence = {
			estate = estate_mamluks
			influence = 80
		}
		OR = {
			NOT = { has_country_flag = mamluks_estate_triggered }
			had_country_flag = {
				flag = mamluks_estate_triggered
				days = 365
			}
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_mamluks
			influence = 100
		}
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_mamluks
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_mamluks
					influence = 100
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_mamluks
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_mamluks
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_mamluks
					loyalty = 30
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_mamluks
					loyalty = 20
				}
			}
			is_at_war = no
		}
	}
	
	can_end = {
		always = yes
	}

	on_start = estate_mamluks_disaster.1
	
	on_start_effect = {
		custom_tooltip = estate_mamluks_disaster_tooltip
	}
	
	on_monthly = {
	}
}
