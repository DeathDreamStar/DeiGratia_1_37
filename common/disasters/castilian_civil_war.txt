#Castilian Civil War
#A mix of the civil wars with outside intervention and noble families siding against the crown between 1419 and 1479.

castilian_civil_war = {
	potential = {
		tag = CAS
		NOT = { has_country_flag = had_cas_civil_war }
		is_free_or_tributary_trigger = yes
		exists = ARA
		exists = POR
		ARA = {
			OR = {
				is_free_or_tributary_trigger = yes
				AND = {
					is_lesser_in_union = yes
					is_subject_of = CAS
				}
			}
		}
		POR = {
			is_free_or_tributary_trigger = yes
		}
		government = monarchy
		normal_or_historical_nations = yes
		uses_doom = no
	}


	can_start = {
		current_age = age_of_discovery
		is_at_war = no
		has_any_disaster = no
		is_lesser_in_union = no
		custom_trigger_tooltip = {
			tooltip = castilian_civil_war_start.tooltip
			has_ruler_flag = isabel_de_trastamara
			POR = {
				has_consort_flag = joanna_la_beltraneja
			}
		}
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			is_lesser_in_union = yes
			NOT = { government = monarchy }
			custom_trigger_tooltip = {
				tooltip = joanna_la_beltraneja_dead.tooltip
				POR = {
					NOT = {
						has_consort_flag = joanna_la_beltraneja
					}
				}
			}
		}
	}
	
	
	
	
	progress = {
		modifier = {
			factor = 100
			custom_trigger_tooltip = {
				tooltip = joanna_la_beltraneja.tooltip
				POR = {
					has_consort_flag = joanna_la_beltraneja
				}
			}
		}
	}




	
	can_end = {
		NOT = {
			exists = AVZ
		}
	}
	



	modifier = {
		legitimacy = -1
	}	
	
	
	on_start = castilian_civil_war.1
	on_end = castilian_civil_war.100
	
	on_monthly = {
		events = {
		}
	}
}
