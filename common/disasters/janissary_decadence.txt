janissary_decadence = {
	potential = {
		has_estate = estate_janissaries
		estate_influence = {
			estate = estate_janissaries
			influence = 80
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_janissaries
			influence = 100
		}
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_janissaries
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 0.5
			NOT = {
				estate_influence = {
					estate = estate_janissaries
					influence = 100
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_janissaries
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_janissaries
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_janissaries
					loyalty = 30
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_janissaries
					loyalty = 20
				}
			}
			is_at_war = no
		}
	}
	
	can_end = {
		OR = {
			custom_trigger_tooltip = {
				tooltip = force_janissaries_to_undergo_yoklama
				has_country_flag = yoklama
			}
			AND = {
				custom_trigger_tooltip = {
					tooltip = had_disaster_for_five_years
					had_country_flag = { flag = had_janissary_decadence days = 1825 }
				}
				stability = 2
				NOT = { num_of_rebel_controlled_provinces = 1 }
				NOT = { num_of_rebel_armies = 1 }
			}
		}
	}
	
	modifier = {
		legitimacy = -1
		technology_cost = 0.25
		idea_cost = 0.25
	}

	on_start = janissary_decadence.1
	on_end = janissary_decadence.100
	
	on_monthly = {
		events = {
		}
		
		random_events = { 
			1250 = 0
			10 = janissary_decadence.2 #Ruler Killed
			10 = janissary_decadence.3
			10 = janissary_decadence.4
			10 = janissary_decadence.5
			10 = janissary_decadence.6
			10 = janissary_decadence.7
		}
	}
}
