	potential = {
		current_age = age_of_revolutions
	}
	
	allow = {
		custom_trigger_tooltip = {
			tooltip = edict_famine_relief_tooltip

		}

	}
	
	modifier = {
		local_monthly_devastation = -0.25

	ai_will_do = {

		modifier = {
			factor = 0
			AND = {
				has_state_edict = edict_famine_relief

					NOT = {
						devastation = 1
					}	
				}
			}
		}
		modifier = {
			factor = 5

		}

			factor = 0

				NOT = { has_state_edict = edict_famine_relief }
				all_province_in_state = {
					NOT = {
						devastation = 2
					}
				}
			}
		}
		modifier = {
			factor = 5
			
			AND = {
				NOT = { has_state_edict = edict_famine_relief }
				all_province_in_state = {
					devastation = 5
				}
			}
		}
		modifier = {
			factor = 5
			
			AND = {
				NOT = { has_state_edict = edict_famine_relief }

	}
