tribe_becomes_monarchy_reform = {
	icon = "king_highlighted"
	allow_normal_conversion = yes
	trigger = {
		NOT = { has_government_attribute_short_desc = locked_government_type }
	}
	effect = {
		set_country_flag = populists_in_government
		lose_reforms = 4
		if = {
			limit = {
				technology_group = nomad_group
				OR = {
					religion_group = muslim
					religion_group = zoroastrian_group
					religion = druze
					secondary_religion = sunni
					secondary_religion = shiite
					secondary_religion = ibadi
					secondary_religion = zoroastrian
					secondary_religion = yazidi
					secondary_religion = druze
				}
			}
			change_technology_group = muslim
			change_unit_type = muslim
		}
		if = {
			limit = {
				technology_group = nomad_group
				OR = {
					religion_group = christian
					religion = jewish
				}
			}
			change_technology_group = eastern
			change_unit_type = eastern
		}
		if = {
			limit = {
				technology_group = nomad_group
				religion = totemism
			}
			change_technology_group = north_american
			change_unit_type = north_american
		}
		if = {
			limit = {
				technology_group = nomad_group
				religion = inti
			}
			change_technology_group = andean
			change_unit_type = andean
		}
		if = {
			limit = {
				technology_group = nomad_group
				OR = {
					religion = mesoamerican_religion
					religion = nahuatl
					religion = mixtec_religion
					religion = zapotec_religion
					religion = purepecha_religion
				}
			}
			change_technology_group = mesoamerican
			change_unit_type = mesoamerican
		}
		if = {
			limit = {
				technology_group = nomad_group
				NOT = { religion_group = muslim }
				NOT = { religion_group = zoroastrian_group }
				NOT = { secondary_religion = sunni }
				NOT = { secondary_religion = shiite }
				NOT = { secondary_religion = ibadi }
				NOT = { secondary_religion = zoroastrian }
				NOT = { secondary_religion = yazidi }
				NOT = { secondary_religion = druze }
				NOT = { religion_group = christian }
				NOT = { religion = jewish }
				NOT = { religion = totemism }
				NOT = { religion = inti }
				NOT = { religion = nahuatl }
				NOT = { religion = mesoamerican_religion }
			}
			change_technology_group = chinese
			change_unit_type = chinese
		}
		change_government = monarchy
	}
	ai = {
		factor = 1
		modifier = {
			factor = 2
			OR = {
				religion_group = muslim
				religion_group = eastern
			} 
		}
		modifier = {
			factor = 0
			is_nomad = yes
		}
		modifier = {
			factor = 0
			is_march = yes
		}
	}
}
