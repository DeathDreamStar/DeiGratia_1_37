# First policy below is the default if no other selected

maximize_profit =
{
	can_select = {
		OR = {
			FROM = {
				has_trader = ROOT
			}
			custom_trigger_tooltip = {
				tooltip = is_home_trade_node_tooltip
				home_trade_node = {
					province_id = FROM
				}
			}
		}
	}

	trade_power = {
		duration = -1
		power_modifier = 0.05
		key = maximize_profits
	}
	
	center_of_reformation = no
	button_gfx = GFX_Trading_Policy_Max_Profit
}

propagate_religion = {
	unique = yes # Only one country can select this in a certain node.
	
	can_select = {		
		religion_group = muslim
		dominant_religion = ROOT
		OR = {
			FROM = {
				has_trader = ROOT
			}
			custom_trigger_tooltip = {
				tooltip = is_home_trade_node_tooltip
				home_trade_node = {
					province_id = FROM
				}
			}
		}
		FROM = {
			is_node_in_trade_company_region = yes
			trade_share = {
				country = ROOT
				share = 10
			}
			custom_trigger_tooltip = {
				tooltip = is_not_indian_trade_node_tooltip
				NOT = {
					province_id = 517 # Gujarat
					province_id = 507 # Lahore
					province_id = 524 # Doab
					province_id = 1343 # Bengal
					province_id = 1948 # Deccan
					province_id = 1340 # Coromandel
				}
			}
		}
	}

	can_maintain = {
		religion_group = muslim
		dominant_religion = ROOT
		OR = {
			FROM = {
				has_trader = ROOT
			}
			custom_trigger_tooltip = {
				tooltip = is_home_trade_node_tooltip
				home_trade_node = {
					province_id = FROM
				}
			}
		}
		FROM = {
			is_node_in_trade_company_region = yes
			trade_share = {
				country = ROOT
				share = 5
			}
			custom_trigger_tooltip = {
				tooltip = is_not_indian_trade_node_tooltip
				NOT = {
					province_id = 517 # Gujarat
					province_id = 507 # Lahore
					province_id = 524 # Doab
					province_id = 1343 # Bengal
					province_id = 1948 # Deccan
					province_id = 1340 # Coromandel
				}
			}
		}
	}

	show_alert = yes
	center_of_reformation = yes
	button_gfx = GFX_Trading_Policy_Propagate_Religion
}
