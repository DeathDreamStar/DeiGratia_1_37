	RecruitNewSettlers = {
		major = yes
		potential = {
			government = native
			any_owned_province = {
				is_city = yes
				any_empty_neighbor_province = {
					tribal_land_of = ROOT
				}
			}
		}
		allow = {
			adm_tech = 3
			is_at_war = no
			any_owned_province = {
				is_city = yes
				any_empty_neighbor_province = {
					tribal_land_of = ROOT
				}
			}
			NOT = {
				has_country_modifier = recently_recruited_settlers
			}
		}
		effect = {
			random_owned_province = {
				limit = {
					is_city = yes
					any_empty_neighbor_province = {
						tribal_land_of = ROOT
					}
				}
				random_empty_neighbor_province = {
					limit = {
						tribal_land_of = ROOT
					}
					settle_province = ROOT
				}
			}
			add_country_modifier = {
				name = recently_recruited_settlers
				duration = 3650
			}
		}
		ai_will_do = {
			factor = 1
		}
